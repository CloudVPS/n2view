  5 OP.data.classes.model.OPClass = Backbone.Model.extend({
  6     initialize : function () {
  7        
  8     },
  9     
 10     context : null,
 11     
 12     setContext : function (context) {
 13         this.context = context;
 14         this.setStore(context.getOpClassStore());
 15     },
 16     
 17     getContext : function () {
 18         return this.context;
 19     },
 20     
 21     /**
 22      * @returns {boolean}
 23      */
 24     isNew : function () {
 25         return !this.get("uuid");
 26     },
 27     
 28     store : null,
 29     setStore : function (store){
 30         this.store = store;
 31     },
 32     getStore : function () {
 33         return this.store;
 34     },
 35         
 36     defaults : {
 37          canCreate: false,
 38          canDelete: false,
 39          canGetInfo: false,
 40          canUpdate: false,
 41          childCount: 0,
 42          description: "",
 43          indexing: "",
 44          isRootObject: false,
 45          meta: false,
 46          metaValue: "",
 47          name: "",
 48          parent: null,
 49          singleton: false,
 50          singletonValue: "",
 51          title: "",
 52          uuid: null,
 53          parentId : null
 54      },
 55      
 56      /**
 57       * 
 58       * @param parentId
 59       * @returns {OP.data.classes.collection.OPObject}
 60       */
 61      getOPObjectCollection : function (parentId) {
 62          var opObjectCollection = new OP.data.classes.collection.OPObject([], { opClass : this, parentId : parentId != undefined ? parentId : null });
 63          opObjectCollection.setContext(this.context);
 64          return opObjectCollection;
 65      },
 66      
 67      /**
 68       * 
 69       * @param options
 70       * @returns {OP.data.classes.model.OPObject}
 71       */
 72      getOPObject : function (options){
 73          var opObject = new OP.data.classes.model.OPObject(options);
 74          opObject.setContext(this.context);
 75          opObject.opClass = this;
 76          return opObject;
 77      },
 78      
 79      /**
 80       * @function
 81       * @param uuid
 82       * @returns {OP.data.classes.model.OPObject}
 83       */
 84      getOPObjectByUUID : function (uuid) {
 85          return this.getOPObject({ id : uuid });
 86      },
 87      
 88      /**
 89       * @param metaId
 90       * @returns {OP.data.classes.model.OPObject}
 91       */
 92      getOPObjectByMetaId : function (metaId) {
 93          return this.getOPObject({ id : metaId });
 94      }
 95  });
