  6 OP.data = {
  7     
  8         initialize : function () {
  9         Backbone.sync = function (method, model, success, error){
 10            // console.log(method, model, success, error);
 11             switch (method) {
 12                 case "create":         model.store.create(model, success, error); break;
 13                 case "read":           model.id ? model.store.find(model, success, error) : model.store.findAll(model, success, error); break;
 14                 case "update":         model.store.update(model, success, error); break;
 15                 case "delete":         model.store.destroy(model); break;
 16             }
 17         };
 18         
 19         OP.data.classes.collection.OPClass.model = OP.data.classes.model.OPClass;
 20         OP.data.classes.collection.OPObject.model = OP.data.classes.model.OPObject;
 21     },
 22     
 23     contexts : []
 24 };
