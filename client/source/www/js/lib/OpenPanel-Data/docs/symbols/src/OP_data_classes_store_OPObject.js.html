  5 OP.data.classes.store.OPObject = function (sessionId) {
  6     this.sessionId = sessionId;
  7     this.data = {};
  8     this.hasFetched = false;
  9     this.openPanelClassCollection = null;
 10     this.objectsByUuid = {};
 11     _.extend(this, {
 12         find : function (model, success, error) {
 13             var that = this;
 14             OP.rpc.requests.getRecord({
 15                 sessionId : model.getContext().getSessionId(),
 16                 classId : model.opClass.id,
 17                 objectId : model.id,
 18                 success : function (attributes) {
 19                     success(attributes);
 20                 },
 21                 error : function () {
 22                     error(arguments);
 23                 }
 24             });
 25         },
 26         create : function (model, success, error) {
 27             error("OPClassCollection can not be created this way." 
 28                     +" See http://wiki.openpanel.com/index.php/Developers");
 29         },
 30         update : function (model, success, error) {
 31             error("OPClassCollection can not be updated this way.");
 32         },
 33         destroy : function (model, success, error) {
 34             error("OPClassCollection can not be deleted this way.");
 35         },
 36         findAll : function (collection, success, error) {
 37             var that = this;
 38             var target = collection.parentId != null ? collection.parentId : "ROOT";
 39             if(collection.opClass.id == "ROOT") target = "_";
 40             this.data[target] = {};
 41             if(target == "_") {
 42                 that.data[target][collection.opClass.id] = {};
 43                 var o = new OP.data.classes.model.OPObject({ id : "ROOT", uuid : "ROOT"});
 44                 o.setContext(collection.getContext());
 45                 o.opClass = collection.opClass;
 46                 that.data[target][collection.opClass.id]["ROOT"] = o;
 47                 success(_.values(that.data[target][collection.opClass.id]));
 48             } else {
 49                 OP.rpc.requests.getRecords(
 50                     {
 51                         sessionId : collection.getContext().getSessionId(), 
 52                         classId : collection.opClass.id, 
 53                         parentId : collection.parentId != null ? collection.parentId : undefined,
 54                         success : function (records, response){
 55                             _.each(records, function (record, id) {
 56                                 if(that.data[target][collection.opClass.id] == undefined) {
 57                                     that.data[target][collection.opClass.id] = {};
 58                                 }
 59                                 var o = new OP.data.classes.model.OPObject(record);
 60                                 o.opClass = collection.getContext().classCollection.get(record["class"]);
 61                                 o.setContext(collection.getContext());
 62                                 that.data[target][collection.opClass.id][record.uuid] = o;
 63                             });
 64                             success(_.values(that.data[target][collection.opClass.id]));
 65                         },
 66                         error : function (errorMessage, request, response){ 
 67                             error(errorMessage);
 68                         }
 69                     }
 70                 );
 71             }
 72         }
 73     });
 74 };
