 18     /** @function
 19      * 
 20      * @param url
 21      * @param requestString
 22      * @param httpMethod
 23      * @param successCallback
 24      * @param errorCallback
 25      */
 26     clientRequestHandler : function (url, requestString, httpMethod, successCallback, errorCallback) {
 27         var xmlHttp = false;
 28         try {
 29             xmlHttp = new ActiveXObject("Msxml2.XMLHTTP");
 30         } catch (e1) {
 31             try {
 32                 xmlHttp = new ActiveXObject("Microsoft.XMLHTTP");
 33             } catch (anotherError) {
 34                 xmlHttp = false;
 35             }
 36         }
 37         
 38         if (!xmlHttp && typeof XMLHttpRequest !== "undefined") {
 39             try {
 40                 xmlHttp = new XMLHttpRequest();
 41             } catch (e2) {
 42                 xmlHttp = false;
 43             }
 44         }
 45         if (!xmlHttp && window.createRequest) {
 46             try {
 47                 xmlHttp = window.createRequest();
 48             } catch (e3) {
 49                 xmlHttp = false;
 50             }
 51         }
 52 
 53         xmlHttp.open(httpMethod, url, true);
 54         
 55         xmlHttp.onreadystatechange = function () {
 56             if (xmlHttp.readyState === 4) {
 57                 if (xmlHttp.status === 200 || xmlHttp.status === 0) {
 58                     successCallback(JSON.parse(xmlHttp.responseText));
 59                 } else {
 60                     errorCallback(xmlHttp);
 61                 }
 62             }
 63         };
 64         xmlHttp.send(requestString);
 65     },
 66     /** @function
 67      * 
 68      * @param url
 69      * @param requestString
 70      * @param httpMethod
 71      * @param successCallback
 72      * @param errorCallback
 73      */
 74     serverRequestHandler : function (url, requestString, httpMethod, successCallback, errorCallback) {
 75         var http = require('http');
 76         var parsedUrl = require('url').parse(url);
 77         var remote_client = http.createClient(parsedUrl.port ? parsedUrl.port : 80, parsedUrl.hostname);
 78         var request = remote_client.request("POST", parsedUrl.pathname, {"host": parsedUrl.hostname, "user-agent": "node.js"});
 79         var body = "";
 80         request.addListener('response', function (response) {
 81             response.setEncoding('binary'); 
 82 
 83             response.addListener('data', function (chunk) {
 84                 body += chunk;
 85             });
 86 
 87             response.addListener('end', function () {
 88                 if (response.statusCode === 200) {
 89                     successCallback(JSON.parse(body));
 90                 } else {
 91                     errorCallback(body);
 92                 }
 93             });
 94         });
 95         request.write(requestString);
 96         request.end();
 97     },

126     /** Validates response 
127         @param {Object} jsonData A chunk of response data
128         @param {Object} jsonSchema A chunk of validation schema
129         @param {String} requestString A string containing the request in json format
130     */
131     validateResponse : function (jsonData, jsonSchema, requestString) {
132         if (this.jsvEnvironment === null) {
133             this.jsvEnvironment = JSV.createEnvironment();
134         }
135         var report = this.jsvEnvironment.validate(jsonData, jsonSchema);
136         var errorMessage;
137         var errorMessages;
138         
139         if (report.errors.length !== 0) {
140             errorMessages = [];
141             for (var i = 0;i < report.errors.length;i++) {
142                 errorMessages.push("[" + i + "] " + report.errors[i].message + " (" + report.errors[i].schemaUri + ")");
143             }
144 
145             if (requestString !== null) {
146                 errorMessage = "Response from request " + requestString + " has issues:\n\t%s" + errorMessages.join(",\n\t");
147             } else {
148                 errorMessage = errorMessages.join(",\n");
149             }
150         }
151         
152         return errorMessage;
153     },
154 
155     findObjectPath : function (subject, path, messages) {
156         if (messages === undefined) {
157             messages = {};
158         }
159         var i = 0, a = subject, r = path.split(".");
160         
161         while (a[r[i]]) {
162             a = a[r[i++]];
163         }
164 
165         if (i < r.length) {
166             messages.error = "Can't find: " + r.splice(0, i + 1).join(".");
167         } else {
168             return a;
169         }
170     },

172     asynchronousRequest : function (options) {
173         if (this.requestHandler === null) {
174             this.setRequestHandler(this.isRunningOnClient === true 
175                     ? this.clientRequestHandler : this.serverRequestHandler);
176         }
177         if (options !== null && options.request !== undefined) {
178             var request = options.request;
179             var returnObjectPath = options.returnObjectPath;
180             var success = options.success;
181             var error = options.error;
182             var schema = options.schema;
183             var requestString = JSON.stringify(request);
184             var that = this;
185             var successCallback = function (data) {
186                 if (!!(success && success.constructor && success.call && success.apply) || !!(error && error.constructor && error.call && error.apply)) {
187                     var errorMessage;
188                     var successData;
189 
190                     var serverErrorId = that.findObjectPath(data, "header.errorid");
191                     if (serverErrorId !== undefined) {
192                         errorMessage = that.findObjectPath(data, "header.error");
193                         if (errorMessage === undefined) {
194                             errorMessage = "(?) Undefi ned error";
195                         }    
196                     } else {
197                         if (options.schema !== undefined) {
198                             errorMessage = that.validateResponse(data, schema, requestString);
199                         }
200                         if (errorMessage === null) {
201                             if (!!(returnObjectPath === '' || (returnObjectPath && returnObjectPath.charCodeAt && returnObjectPath.substr))) {
202                                 var messages = {};
203                                 var resultValue = that.findObjectPath(data, returnObjectPath, messages);
204                                 if (resultValue !== null && resultValue !== undefined) {
205                                     successData = resultValue;
206                                 } else {
207                                     errorMessage = messages.error;
208                                 }
209                             } else {
210                                 successData = data;
211                             }
212                         }
213                     }
214                     
215                     if (errorMessage === null) {
216                         if (!!(success && success.constructor && success.call && success.apply)) {
217                             if (!!(returnObjectPath === '' || (returnObjectPath && returnObjectPath.charCodeAt && returnObjectPath.substr))) {
218                                 success(resultValue, data);
219                             } else {
220                                 success(data);
221                             }
222                         }
223                     } else {
224                         if (!!(error && error.constructor && error.call && error.apply)) {
225                             error(errorMessage, request, data); 
226                         }
227                     }
228                 }
229             };
230             var errorCallback = function (data) {
231                 if (data.status !== 200) {
232                     error("HTTP error: " + data.status + " " + data.statusText);
233                 }
234             };
235             this.requestHandler(this.url, requestString, "POST", successCallback, errorCallback);
236         }
237     }
